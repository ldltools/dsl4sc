\begin{Name}{1}{safeguard}{author}{LDL Tools}{title}
safeguard - LDL-based requirement enforcement tool
\end{Name}

\section{Synopsis}
\Prog{safeguard} <option>* <infile>?

\section{Description}
\Prog{safeguard} reads a program from <infile>
and attach pre/post conditions to event handlers in the program.

\section{Options}

\begin{description}
\item[\Opt{-o} \emph{file}]
  output to \emph{file}
\item[\Opt{--spec\Lbr:lang\Rbr} \Arg{file}]
  read \emph{file} in \emph{lang} as spec (\emph{lang} ::= dsl | dfa | scxml)\\
  when this option is not provided, \Prog{safeguard} tries auto-detection.
\item[\Opt{--code\Lbr:lang\Rbr} \Arg{file}]
  read \emph{file} in \emph{lang} as a program code (\emph{lang} ::= js | ts)\\
  note that '\Prog{safeguard} --code \Arg{file}' is synonymous with '\Prog{safeguard} \Arg{file}'.
%
\item[\Opt{-h}, \Opt{--help}]
  show usage

\end{description}
